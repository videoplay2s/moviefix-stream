            <span id="player-title" class="text-sm font-bold text-red-500 truncate max-w-xs md:max-w-xl">🍿 Loading Real Stream Link...</span>

        <div class="flex flex-col md:flex-row justify-between items-start md:items-center mb-6 gap-4">

            <input type="text" id="search-box" oninput="searchMedia(this.value)" placeholder="Search any movie or series..." class="bg-zinc-900 text-white border border-zinc-700 text-sm rounded px-4 py-2 w-full md:w-64 focus:outline-none focus:border-red-600">

    <script>
        // CONFIGURATION PARAMETERS
        const TMDB_KEY = '844132b4db1bea14ba6748c14d104ee4'; 
        const ADSTERRA_DIRECT_URL = 'https://youtu.be/aNsAsiJs9XA?si=2dFPm3Ni7S_riofj'; // <- ENTER YOUR ADSTERRA DIRECT LINK HERE
        const STREAM_BASE = 'https://vidsrc.to'; 

        let currentActiveTab = 'movie';

        // Auto Fetch New Releases dynamically matching Netflix Network updates
        async function fetchNetflixMedia(type = 'movie') {
            currentActiveTab = type;
            document.getElementById('grid-title').innerText = type === 'movie' ? '🎬 New Netflix Movies (Auto-Updated)' : '📺 Trending Netflix TV Series';
            
            // TMDB query parameters targeting popular network streaming items automatically
            const apiEndpoint = `https://themoviedb.org{type}?api_key=${TMDB_KEY}&sort_by=popularity.desc&with_networks=213`;
            
            try {
                const response = await fetch(apiEndpoint);
                const data = await response.json();
                renderContentGrid(data.results);
            } catch (err) {
                console.error("Network Stream Sync failed:", err);
            }
        }

        // Render Dynamic Database content into UI Cards
        function renderContentGrid(items) {
            const container = document.getElementById('media-grid');
            container.innerHTML = '';

            if(!items || items.length === 0) {
                container.innerHTML = '<p class="text-zinc-500 text-sm col-span-full text-center">No streaming files found. Try searching another term.</p>';
                return;
            }

            items.forEach(media => {
                const name = media.title || media.name;
                const path = media.poster_path ? `https://tmdb.org{media.poster_path}` : 'https://placeholder.com';
                const year = (media.release_date || media.first_air_date || '2026-01-01').split('-')[0];

                const block = document.createElement('div');
                block.className = 'movie-card relative bg-zinc-900 rounded overflow-hidden shadow-md cursor-pointer transform hover:scale-105 transition duration-200 border border-zinc-800';
                block.onclick = () => triggerRealStream(media.id, name);

                block.innerHTML = `
                    <div class="relative pt-[145%]">
                        <img src="${path}" alt="${name}" class="absolute top-0 left-0 w-full h-full object-cover">
                        <div class="play-overlay absolute inset-0 bg-black/70 flex flex-col items-center justify-center opacity-0 transition duration-200">
                            <span class="bg-red-600 text-white px-4 py-2 rounded-full text-xs font-bold uppercase tracking-wider shadow-lg">▶ Stream HD</span>
                        </div>
                    </div>
                    <div class="p-2.5">
                        <h3 class="font-semibold text-xs truncate text-zinc-100">${name}</h3>
                        <div class="flex justify-between items-center mt-1 text-[10px] text-zinc-400">
                            <span>${year}</span>
                            <span class="text-yellow-500 font-bold">⭐ ${media.vote_average.toFixed(1)}</span>
                        </div>
                    </div>
                `;
                container.appendChild(block);
            });
        }

        // Search engine optimization mapping
        async function searchMedia(query) {
            if(query.trim().length === 0) {
                fetchNetflixMedia(currentActiveTab);
                return;
            }
            try {
                const response = await fetch(`https://themoviedb.org{currentActiveTab}?api_key=${TMDB_KEY}&query=${encodeURIComponent(query)}`);
                const data = await response.json();
                renderContentGrid(data.results);
            } catch(e) {
                console.error("Search index failed", e);
            }
        }

        // Main Stream Activation and Monetization Router
        function triggerRealStream(mediaId, mediaTitle) {
            // Monetization Layer: Instant Popup routing to High-CPM AdSterra Zone
            window.open(ADSTERRA_DIRECT_URL, '_blank');

            // Video Playback Engine Integration 
            const targetSection = document.getElementById('player-section');
            const targetFrame = document.getElementById('stream-frame');
            const targetTitle = document.getElementById('player-title');

            // Route dynamically depending on movie or serial structure (Season 1 Episode 1 default trigger for serials)
            const finalStreamUrl = currentActiveTab === 'movie' ? `${STREAM_BASE}movie/${mediaId}` : `${STREAM_BASE}tv/${mediaId}/1/1`;

            targetFrame.src = finalStreamUrl;
            targetTitle.innerText = `🍿 Currently Playing: ${mediaTitle} (Multi-source High Speed Player)`;
            targetSection.classList.remove('hidden');

            // Focus screen view on active frame
            targetSection.scrollIntoView({ behavior: 'smooth' });
        }

        function closePlayer() {
            const targetSection = document.getElementById('player-section');
            const targetFrame = document.getElementById('stream-frame');
            targetFrame.src = '';
            targetSection.classList.add('hidden');
        }

        // Initialize script sequence
        window.onload = () => fetchNetflixMedia('movie');
    </script>
